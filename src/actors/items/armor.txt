ACTOR ArgArmor : BasicArmorBonus
{
	Inventory.Icon "BON2B0"
	Armor.Savepercent 100
	Armor.Saveamount 1
	Armor.Maxsaveamount 150
}

ACTOR ArgArmorBonus : CustomInventory replaces ArmorBonus
{
	Radius 20
	Height 16
	Scale 0.66
	Inventory.PickupSound "items/shard"
	Inventory.Pickupmessage "$GOTARMBONUS"
	+RANDOMIZE
	+COUNTITEM
	+INVENTORY.ALWAYSPICKUP
	States
	{
	Spawn:
		BON2 A 30
		BON2 A 5 BRIGHT
		loop
	Pickup:
	    TNT1 A 0 ACS_NamedExecuteWithResult("Orange_GiveArmor", 5)
	    Stop
	}
}
